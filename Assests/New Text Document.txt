pragma solidity ^0.4.22;

contract Customer{
    
    int private cid;
    string private name;
    string private mobile;
    int private balance;
    
    constructor(int _cid, string _name, string _mobile, int _balance) public
    {
        cid = _cid;
        name = _name;
        mobile = _mobile;
        balance = _balance;
        
        
    }
    
    function getCid() public view returns(int) {
        return cid;
    }
    
     function getName() public view returns(string) {
        return name;
    }
    
    function getMobile() public view returns(string) {
        return mobile;
    }
    
     function getBalance() public view returns(int) {
        return balance;
    }
    
    function setMobile(string _mobile) public {
        mobile = _mobile;
    }
    
    function deposit(int _amount) public {
        balance = balance + _amount;
    }
    
     function withraw(int _amount) public {
        balance = balance - _amount;
    }
}
